Timer.prototype.applyEasing = function() {
  var t = this.percentage;
  if (this.easing === 'ease-in') {
    return t * t;
  } else if (this.easing === 'ease-out') {
    return t * (2 - t);
  } else if (this.easing === 'ease-in-out') {
    if (t < 0.5) {
      return 2 * t * t;
    } else {
      return -1 + (4 - 2 * t) * t;
    }
  }
  return t; // default is linear.
};
